  2"""
  3Inventory application for a tech video channel featuring computer builds.
  4Channel has a pool of inventory, (for example 5 x AMD Ryzen 2-2700 CPUs) used for builds.
  5When a CPU is taken from the pool, it will be indicated using the object that tracks that 
  6specific type of CPU. Channel may also purchase additional CPUs, or retire some.
  7"""

 10class Resources:
 11    """Resources base class"""
 12
 13    def __init__(self, name, manufacturer, total, allocated):
 14        """
 15        Args:
 16           name (str): user-friendly name of resource instance (e.g.` Intel Core i9-9900K`)
 17           manufacturer (str): resource instance manufacturer (e.g. `Nvidia`)
 18           total (int): inventory total (how many are in the inventory pool)
 19           allocated (int): number allocated (how many are already in use), must be smaller than total
 20        """
 21        self._name = set_if_valid_string(name)
 22        self._manufacturer = set_if_valid_string(manufacturer)
 23        self._total = set_if_valid_int(total)
 24        # Initial allocated cannot be bigger than the total
 25        if self._total - set_if_valid_int(allocated) >= 0:
 26            self._allocated = set_if_valid_int(allocated)
 27        else:
 28            raise ValueError
 29
 30    # Total and allocated should be read-only
 31    # modifiable through the various methods such as `claim`, `return`, `died` and `purchased`
 32    @property
 33    def total(self):
 34        """
 35        Returns:
 36            total (int): total amount of resource
 37        """
 38        return self._total
 39
 40    @property
 41    def allocated(self):
 42        """
 43        Returns:
 44            allocated (int): amount of allocated resource
 45        """
 46        return self._allocated
 47
 48    # name and manufacturer should be read-only
 49    @property
 50    def name(self):
 51        """
 52            Returns:
 53                name(str):resource name
 54        """
 55        return self._name
 56
 57    @property
 58    def manufacturer(self):
 59        """
 60            Returns:
 61                manufacturer(str):resource manufacturer
 62        """
 63        return self._manufacturer
 64
 65    def __repr__(self) -> str:
 66        return (f"\nResource: {self.name}\nManufacturer: {self.manufacturer}\nTotal available: {self.total}\n" +
 67                f"Allocated resource: {self._allocated}")
 68
 69    def __str__(self) -> str:
 70        return self._name
 71
 72    def claim(self, n):
 73        """
 74        Claim n inventory items if available
 75        Args:
 76            n(int): number of items to claim
 77        Returns:
 78            None
 79        """
 80        if self._total - n >= 0:
 81            self._allocated += set_if_valid_int(n)
 82        else:
 83            raise ValueError(
 84                "Allocated resources cannot exceed total resources")
 85
 86    def freeup(self, n):
 87        """
 88        Free up n items if possible
 89        Args:
 90            n(int): number of items to free up
 91        Returns:
 92            None
 93        """
 94        if self._allocated - n >= 0:
 95            self._allocated -= set_if_valid_int(n)
 96        else:
 97            raise ValueError(
 98                "Cannot free up more resources than allocated")
 99
100    def died(self, n):
101        """
102        Subtract n items from total and allocated (if possible)
103        Args:
104            n(int): number of items to subtract
105        Returns:
106            None
107        """
108        if set_if_valid_int(self.total - set_if_valid_int(n)):
109            self._total -= set_if_valid_int(n)
110            self._allocated -= set_if_valid_int(n)
111
112    def purchased(self, n):
113        """
114        Add n items to the total 
115        Args:
116            n(int): number of items to add
117        Returns:
118            None
119        """
120        if set_if_valid_int(n):
121            self._total += set_if_valid_int(n)
122
123    @property
124    def category(self):
125        """
126            Returns:
127                category(str):resource category
128        """
129        return (str(type(self)).split(".")[-1][:-2]).lower()
